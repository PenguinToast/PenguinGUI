Class CheckBox

A check box.

Usage:

     -- Create a checkbox that prints when it is checked
 local checkbox = CheckBox(0, 0, 16)
 checkbox:addListener("selected", function(t, k, old, new)
   print("The checkbox was " .. (new and "checked" or "unchecked"))
 end
